import java.util.Scanner; public class Main { public static int longestSequenceOfOnes(int binaryNumber) { String binaryString = Integer.toBinaryString(binaryNumber); int count = 0; int maxCount = 0; for (char digit : binaryString.toCharArray()) { if (digit == '1') { count++; if (count > maxCount) { maxCount = count; } } else { count = 0; } } return maxCount; } public static int longestConsecutiveOnes(int n) { String binary = Integer.toBinaryString(n); int maxLength = 0; int currentLength = 0; int previousLength = 0; for (char bit : binary.toCharArray()) { if (bit == '1') { currentLength++; } else { maxLength = Math.max(maxLength, currentLength + previousLength + 1); previousLength = currentLength; currentLength = 0; } } maxLength = Math.max(maxLength, currentLength + previousLength + 1); return maxLength; } public static void main(String[] args) { Scanner scanner = new Scanner(System.in); System.out.print("Enter a regular number: "); int number = scanner.nextInt(); int binaryNumber = Integer.parseInt(Integer.toBinaryString(number), 2); int longestSequence = longestSequenceOfOnes(binaryNumber); System.out.println( "The longest sequence of 1's in the binary representation of " + number + " is: " + longestSequence); int longestConsecutive = longestConsecutiveOnes(binaryNumber); System.out.println("The longest consecutive sequence of 1's in the binary representation of " + number + " is: " + longestConsecutive); scanner.close(); } }
