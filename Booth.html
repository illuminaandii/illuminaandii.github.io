import java.util.Scanner; public class Main { public int multiply(int n1, int n2) { int m = n1; int r = n2; int A = n1; int S = -n1; int P = 0; int count = Integer.SIZE; while (count > 0) { if ((r & 1) == 1) { P += A; S += m; } A <<= 1; S <<= 1; count--; r >>= 1; } return P; } public static void main(String[] args) { Scanner scan = new Scanner(System.in); Main b = new Main(); System.out.println("Enter two binary numbers (without '0b' prefix) -"); String binary1 = scan.next(); String binary2 = scan.next(); int n1 = Integer.parseInt(binary1, 2); int n2 = Integer.parseInt(binary2, 2); int result = b.multiply(n1, n2); System.out.println("\n\nResult : " + binary1 + " * " + binary2 + " = " + Integer.toBinaryString(result)); } }
