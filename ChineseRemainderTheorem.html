import java.util.*; class ChineseRemainder { int calculate(int size, int div[], int rem[]) { int x = 1; while (true) { int j; for (j = 0; j < size; j++) { if (x % div[j] != rem[j]) break; } if (j == size) return x; x++; } } } public class Main { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter the number of congruences: "); int size = sc.nextInt(); int[] div = new int[size]; int[] rem = new int[size]; System.out.println("Enter Divisors and Remainders (space-separated):"); for (int i = 0; i < size; i++) { div[i] = sc.nextInt(); rem[i] = sc.nextInt(); } ChineseRemainder c = new ChineseRemainder(); int result = c.calculate(size, div, rem); System.out.println("The smallest possible solution is: " + result); } }
